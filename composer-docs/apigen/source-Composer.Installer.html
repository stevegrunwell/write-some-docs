    238: // Disable GC to save CPU cycles, as the dependency solver can create hundreds of thousands
    239: // of PHP objects, the GC can spend quite some time walking the tree of references looking
    240: // for stuff to collect while there is nothing to collect. This slows things down dramatically
    241: // and turning it off results in much better performance. Do not try this at home however.
    242: gc_collect_cycles();

    251: // Force update if there is no lock file present

    255: }

    406: } else {

    423: if ($this->io->isVerbose()) {
    424: $this->io->error('['.get_class($e).'] '.$e->getMessage());
    425: }
    426: }
    427: } else {
    428: $this->io->writeError('No '.$target.' packages - skipping audit.');
    429: }
    430: }
    431:
    432: return 0;

    438: protected function doUpdate(InstalledRepositoryInterface $localRepo, bool $doInstall): int
    439: {
    440: $platformRepo = $this->createPlatformRepo(true);
    441: $aliases = $this->getRootAliases(true);
    442:
    443: $lockedRepository = null;
    444:
    445: try {
    446: if ($this->locker->isLocked()) {
    447: $lockedRepository = $this->locker->getLockedRepository(true);
    448: }
    449: } catch (\Seld\JsonLint\ParsingException $e) {
    450: if ($this->updateAllowList !== null || $this->updateMirrors) {
    451: // in case we are doing a partial update or updating mirrors, the lock file is needed so we error
    452: throw $e;
    453: }
    454: // otherwise, ignoring parse errors as the lock file will be regenerated from scratch when
    455: // doing a full update
    456: }
    457:
    458: if (($this->updateAllowList !== null || $this->updateMirrors) && !$lockedRepository) {
    459: $this->io->writeError('<error>Cannot update ' . ($this->updateMirrors ? 'lock file information' : 'only a partial set of packages') . ' without a lock file present. Run `composer update` to generate a lock file.</error>', true, IOInterface::QUIET);
    460:
    461: return self::ERROR_NO_LOCK_FILE_FOR_PARTIAL_UPDATE;
    462: }
    463:
    464: $this->io->writeError('<info>Loading composer repositories with package information</info>');
    465:
    466: // creating repository set
    467: $policy = $this->createPolicy(true);
    468: $repositorySet = $this->createRepositorySet(true, $platformRepo, $aliases);
    469: $repositories = $this->repositoryManager->getRepositories();
    470: foreach ($repositories as $repository) {
    471: $repositorySet->addRepository($repository);
    472: }
    473: if ($lockedRepository) {
    474: $repositorySet->addRepository($lockedRepository);
    475: }
    476:
    477: $request = $this->createRequest($this->fixedRootPackage, $platformRepo, $lockedRepository);
    478: $this->requirePackagesForUpdate($request, $lockedRepository, true);
    479:
    480: // pass the allow list into the request, so the pool builder can apply it
    481: if ($this->updateAllowList !== null) {
    482: $request->setUpdateAllowList($this->updateAllowList, $this->updateAllowTransitiveDependencies);
    483: }
    484:
    485: $pool = $repositorySet->createPool($request, $this->io, $this->eventDispatcher, $this->createPoolOptimizer($policy));
    486:
    487: $this->io->writeError('<info>Updating dependencies</info>');
    488:
    489: // solve dependencies
    490: $solver = new Solver($policy, $pool, $this->io);
    491: try {
    492: $lockTransaction = $solver->solve($request, $this->platformRequirementFilter);
    493: $ruleSetSize = $solver->getRuleSetSize();
    494: $solver = null;
    495: } catch (SolverProblemsException $e) {
    496: $err = 'Your requirements could not be resolved to an installable set of packages.';
    497: $prettyProblem = $e->getPrettyString($repositorySet, $request, $pool, $this->io->isVerbose());
    498:
    499: $this->io->writeError('<error>'. $err .'</error>', true, IOInterface::QUIET);
    500: $this->io->writeError($prettyProblem);
    501: if (!$this->devMode) {
    502: $this->io->writeError('<warning>Running update with --no-dev does not mean require-dev is ignored, it just means the packages will not be installed. If dev requirements are blocking the update you have to resolve those problems.</warning>', true, IOInterface::QUIET);
    503: }
    504:
    505: $ghe = new GithubActionError($this->io);
    506: $ghe->emit($err."\n".$prettyProblem);
    507:
    508: return max(self::ERROR_GENERIC_FAILURE, $e->getCode());
    509: }
    510:
    511: $this->io->writeError("Analyzed ".count($pool)." packages to resolve dependencies", true, IOInterface::VERBOSE);
    512: $this->io->writeError("Analyzed ".$ruleSetSize." rules to resolve dependencies", true, IOInterface::VERBOSE);
    513:
    514: $pool = null;
    515:
    516: if (!$lockTransaction->getOperations()) {
    517: $this->io->writeError('Nothing to modify in lock file');
    518: }
    519:
    520: $exitCode = $this->extractDevPackages($lockTransaction, $platformRepo, $aliases, $policy, $lockedRepository);
    521: if ($exitCode !== 0) {
    522: return $exitCode;
    523: }
    524:
    525: // exists as of composer/semver 3.3.0
    526: if (method_exists('Composer\Semver\CompilingMatcher', 'clear')) { // @phpstan-ignore-line
    527: \Composer\Semver\CompilingMatcher::clear();
    528: }
    529:
    530: // write lock
    531: $platformReqs = $this->extractPlatformRequirements($this->package->getRequires());
    532: $platformDevReqs = $this->extractPlatformRequirements($this->package->getDevRequires());
    533:
    534: $installsUpdates = $uninstalls = [];
    535: if ($lockTransaction->getOperations()) {
    536: $installNames = $updateNames = $uninstallNames = [];
    537: foreach ($lockTransaction->getOperations() as $operation) {
    538: if ($operation instanceof InstallOperation) {
    539: $installsUpdates[] = $operation;
    540: $installNames[] = $operation->getPackage()->getPrettyName().':'.$operation->getPackage()->getFullPrettyVersion();
    541: } elseif ($operation instanceof UpdateOperation) {
    542: // when mirrors/metadata from a package gets updated we do not want to list it as an
    543: // update in the output as it is only an internal lock file metadata update
    544: if ($this->updateMirrors
    545: && $operation->getInitialPackage()->getName() === $operation->getTargetPackage()->getName()
    546: && $operation->getInitialPackage()->getVersion() === $operation->getTargetPackage()->getVersion()
    547: ) {
    548: continue;
    549: }
    550:
    551: $installsUpdates[] = $operation;
    552: $updateNames[] = $operation->getTargetPackage()->getPrettyName().':'.$operation->getTargetPackage()->getFullPrettyVersion();
    553: } elseif ($operation instanceof UninstallOperation) {
    554: $uninstalls[] = $operation;
    555: $uninstallNames[] = $operation->getPackage()->getPrettyName();
    556: }
    557: }
    558:
    559: if ($this->config->get('lock')) {
    560: $this->io->writeError(sprintf(
    561: "<info>Lock file operations: %d install%s, %d update%s, %d removal%s</info>",
    562: count($installNames),
    563: 1 === count($installNames) ? '' : 's',
    564: count($updateNames),
    565: 1 === count($updateNames) ? '' : 's',
    566: count($uninstalls),
    567: 1 === count($uninstalls) ? '' : 's'
    568: ));
    569: if ($installNames) {
    570: $this->io->writeError("Installs: ".implode(', ', $installNames), true, IOInterface::VERBOSE);
    571: }
    572: if ($updateNames) {
    573: $this->io->writeError("Updates: ".implode(', ', $updateNames), true, IOInterface::VERBOSE);
    574: }
    575: if ($uninstalls) {
    576: $this->io->writeError("Removals: ".implode(', ', $uninstallNames), true, IOInterface::VERBOSE);
    577: }
    578: }
    579: }

    584: } else {
