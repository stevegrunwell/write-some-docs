    120: protected function archive(IOInterface $io, Config $config, ?string $packageName, ?string $version, string $format, string $dest, ?string $fileName, bool $ignoreFilters, ?Composer $composer): int
    121: {
    122: if ($composer) {
    123: $archiveManager = $composer->getArchiveManager();
    124: } else {
    125: $factory = new Factory;
    126: $process = new ProcessExecutor();
    127: $httpDownloader = Factory::createHttpDownloader($io, $config);
    128: $downloadManager = $factory->createDownloadManager($io, $config, $httpDownloader, $process);
    129: $archiveManager = $factory->createArchiveManager($config, $downloadManager, new Loop($httpDownloader, $process));
    130: }
    131:
    132: if ($packageName) {
    133: $package = $this->selectPackage($io, $packageName, $version);
    134:
    135: if (!$package) {
    136: return 1;
    137: }
    138: } else {
    139: $package = $this->requireComposer()->getPackage();
    140: }
    141:
    142: $io->writeError('<info>Creating the archive into "'.$dest.'".</info>');
    143: $packagePath = $archiveManager->archive($package, $format, $dest, $fileName, $ignoreFilters);
    144: $fs = new Filesystem;
    145: $shortPath = $fs->findShortestPath(Platform::getCwd(), $packagePath, true);
    146:
    147: $io->writeError('Created: ', false);
    148: $io->write(strlen($shortPath) < strlen($packagePath) ? $shortPath : $packagePath);
    149:
    150: return 0;

    156: protected function selectPackage(IOInterface $io, string $packageName, ?string $version = null)
    157: {
    158: $io->writeError('<info>Searching for the specified package.</info>');
    159:
    160: if ($composer = $this->tryComposer()) {
    161: $localRepo = $composer->getRepositoryManager()->getLocalRepository();
    162: $repo = new CompositeRepository(array_merge([$localRepo], $composer->getRepositoryManager()->getRepositories()));
    163: $minStability = $composer->getPackage()->getMinimumStability();
    164: } else {
    165: $defaultRepos = RepositoryFactory::defaultReposWithDefaultManager($io);
    166: $io->writeError('No composer.json found in the current directory, searching packages from ' . implode(', ', array_keys($defaultRepos)));
    167: $repo = new CompositeRepository($defaultRepos);
    168: $minStability = 'stable';
    169: }
    170:
    171: if ($version !== null && Preg::isMatchStrictGroups('{@(stable|RC|beta|alpha|dev)$}i', $version, $match)) {
    172: $minStability = $match[1];
    173: $version = (string) substr($version, 0, -strlen($match[0]));
    174: }
    175:
    176: $repoSet = new RepositorySet($minStability);
    177: $repoSet->addRepository($repo);
    178: $parser = new VersionParser();
    179: $constraint = $version !== null ? $parser->parseConstraints($version) : null;
    180: $packages = $repoSet->findPackages(strtolower($packageName), $constraint);
    181:
    182: if (count($packages) > 1) {
    183: $versionSelector = new VersionSelector($repoSet);
    184: $package = $versionSelector->findBestCandidate(strtolower($packageName), $version, $minStability);
    185: if ($package === false) {
    186: $package = reset($packages);
    187: }
    188:
    189: $io->writeError('<info>Found multiple matches, selected '.$package->getPrettyString().'.</info>');
    190: $io->writeError('Alternatives were '.implode(', ', array_map(static function ($p): string {
    191: return $p->getPrettyString();
    192: }, $packages)).'.');
    193: $io->writeError('<comment>Please use a more specific constraint to pick a different package.</comment>');
    194: } elseif (count($packages) === 1) {
    195: $package = reset($packages);
    196: $io->writeError('<info>Found an exact match '.$package->getPrettyString().'.</info>');
    197: } else {
    198: $io->writeError('<error>Could not find a package matching '.$packageName.'.</error>');
    199:
    200: return false;
    201: }
    202:
    203: if (!$package instanceof CompletePackageInterface) {
    204: throw new \LogicException('Expected a CompletePackageInterface instance but found '.get_class($package));
    205: }
    206: if (!$package instanceof BasePackage) {
    207: throw new \LogicException('Expected a BasePackage instance but found '.get_class($package));
    208: }
    209:
    210: return $package;
    211: }
