    117: $io = $this->getIO();

    119: $io->writeError('<warning>You are using the deprecated option "--dev". It has no effect and will break in Composer 3.</warning>');
    120: }
    121: if ($input->getOption('no-suggest')) {
    122: $io->writeError('<warning>You are using the deprecated option "--no-suggest". It has no effect and will break in Composer 3.</warning>');
    123: }
    124:
    125: $composer = $this->requireComposer();
    126:
    127: if (!HttpDownloader::isCurlEnabled()) {
    128: $io->writeError('<warning>Composer is operating significantly slower than normal because you do not have the PHP curl extension enabled.</warning>');
    129: }

    131: $packages = $input->getArgument('packages');
    132: $reqs = $this->formatRequirements($input->getOption('with'));
    133:
    134: // extract --with shorthands from the allowlist
    135: if (count($packages) > 0) {
    136: $allowlistPackagesWithRequirements = array_filter($packages, static function ($pkg): bool {
    137: return Preg::isMatch('{\S+[ =:]\S+}', $pkg);
    138: });
    139: foreach ($this->formatRequirements($allowlistPackagesWithRequirements) as $package => $constraint) {
    140: $reqs[$package] = $constraint;
    141: }
    142:
    143: // replace the foo/bar:req by foo/bar in the allowlist
    144: foreach ($allowlistPackagesWithRequirements as $package) {
    145: $packageName = Preg::replace('{^([^ =:]+)[ =:].*$}', '$1', $package);
    146: $index = array_search($package, $packages);
    147: $packages[$index] = $packageName;

    149: }
    150:
    151: $rootPackage = $composer->getPackage();
    152: $rootPackage->setReferences(RootPackageLoader::extractReferences($reqs, $rootPackage->getReferences()));
    153: $rootPackage->setStabilityFlags(RootPackageLoader::extractStabilityFlags($reqs, $rootPackage->getMinimumStability(), $rootPackage->getStabilityFlags()));
    154:

    254: private function getPackagesInteractively(IOInterface $io, InputInterface $input, OutputInterface $output, Composer $composer, array $packages): array
    255: {
    256: if (!$input->isInteractive()) {
    257: throw new \InvalidArgumentException('--interactive cannot be used in non-interactive terminals.');
    258: }
    259:
    260: $requires = array_merge(
    261: $composer->getPackage()->getRequires(),
    262: $composer->getPackage()->getDevRequires()
    263: );
    264: $autocompleterValues = [];
    265: foreach ($requires as $require) {
    266: $target = $require->getTarget();
    267: $autocompleterValues[strtolower($target)] = $target;
    268: }
    269:
    270: $installedPackages = $composer->getRepositoryManager()->getLocalRepository()->getPackages();
    271: foreach ($installedPackages as $package) {
    272: $autocompleterValues[$package->getName()] = $package->getPrettyName();
    273: }
    274:
    275: $helper = $this->getHelper('question');
    276: $question = new Question('<comment>Enter package name: </comment>', null);
    277:
    278: $io->writeError('<info>Press enter without value to end submission</info>');
    279:
    280: do {
    281: $autocompleterValues = array_diff($autocompleterValues, $packages);
    282: $question->setAutocompleterValues($autocompleterValues);
    283: $addedPackage = $helper->ask($input, $output, $question);
    284:
    285: if (!is_string($addedPackage) || empty($addedPackage)) {
    286: break;
    287: }
    288:
    289: $addedPackage = strtolower($addedPackage);
    290: if (!in_array($addedPackage, $packages)) {
    291: $packages[] = $addedPackage;
    292: }
    293: } while (true);
    294:
    295: $packages = array_filter($packages);
    296: if (!$packages) {
    297: throw new \InvalidArgumentException('You must enter minimum one package.');
    298: }
    299:
    300: $table = new Table($output);
    301: $table->setHeaders(['Selected packages']);
    302: foreach ($packages as $package) {
    303: $table->addRow([$package]);
    304: }
    305: $table->render();
    306:
    307: if ($io->askConfirmation(sprintf(
    308: 'Would you like to continue and update the above package%s [<comment>yes</comment>]? ',
    309: 1 === count($packages) ? '' : 's'
    310: ))) {
    311: return $packages;
    312: }
    313:
    314: throw new \RuntimeException('Installation aborted.');
    315: }
