    15: /**
    16: * The RuleWatchGraph efficiently propagates decisions to other rules
    17: *
    18: * All rules generated for solving a SAT problem should be inserted into the
    19: * graph. When a decision on a literal is made, the graph can be used to
    20: * propagate the decision to all other rules involving the literal, leading to
    21: * other trivial decisions resulting from unit clauses.
    22: *
    23: * @author Nils Adermann <naderman@naderman.de>
    24: */
    25: class RuleWatchGraph
    26: {
    27: /** @var array<int, RuleWatchChain> */
    28: protected $watchChains = [];
    29:
    30: /**
    31: * Inserts a rule node into the appropriate chains within the graph
    32: *
    33: * The node is prepended to the watch chains for each of the two literals it
    34: * watches.
    35: *
    36: * Assertions are skipped because they only depend on a single package and
    37: * have no alternative literal that could be true, so there is no need to
    38: * watch changes in any literals.
    39: *
    40: * @param RuleWatchNode $node The rule node to be inserted into the graph
    41: */
    42: public function insert(RuleWatchNode $node): void
    43: {
    44: if ($node->getRule()->isAssertion()) {
    45: return;
    46: }
    47:
    48: if (!$node->getRule() instanceof MultiConflictRule) {
    49: foreach ([$node->watch1, $node->watch2] as $literal) {
    50: if (!isset($this->watchChains[$literal])) {
    51: $this->watchChains[$literal] = new RuleWatchChain;
    52: }
    53:
    54: $this->watchChains[$literal]->unshift($node);
    55: }
    56: } else {
    57: foreach ($node->getRule()->getLiterals() as $literal) {
    58: if (!isset($this->watchChains[$literal])) {
    59: $this->watchChains[$literal] = new RuleWatchChain;
    60: }
    61:
    62: $this->watchChains[$literal]->unshift($node);
    63: }
    64: }
    65: }
    66:
    67: /**
    68: * Propagates a decision on a literal to all rules watching the literal
    69: *
    70: * If a decision, e.g. +A has been made, then all rules containing -A, e.g.
    71: * (-A|+B|+C) now need to satisfy at least one of the other literals, so
    72: * that the rule as a whole becomes true, since with +A applied the rule
    73: * is now (false|+B|+C) so essentially (+B|+C).
    74: *
    75: * This means that all rules watching the literal -A need to be updated to
    76: * watch 2 other literals which can still be satisfied instead. So literals
    77: * that conflict with previously made decisions are not an option.
    78: *
    79: * Alternatively it can occur that a unit clause results: e.g. if in the
    80: * above example the rule was (-A|+B), then A turning true means that
    81: * B must now be decided true as well.
    82: *
    83: * @param int $decidedLiteral The literal which was decided (A in our example)
    84: * @param int $level The level at which the decision took place and at which
    85: * all resulting decisions should be made.
    86: * @param Decisions $decisions Used to check previous decisions and to
    87: * register decisions resulting from propagation
    88: * @return Rule|null If a conflict is found the conflicting rule is returned
    89: */
    90: public function propagateLiteral(int $decidedLiteral, int $level, Decisions $decisions): ?Rule
    91: {
    92: // we invert the decided literal here, example:
    93: // A was decided => (-A|B) now requires B to be true, so we look for
    94: // rules which are fulfilled by -A, rather than A.
    95: $literal = -$decidedLiteral;
    96:
    97: if (!isset($this->watchChains[$literal])) {
    98: return null;
    99: }
    100:
    101: $chain = $this->watchChains[$literal];
    102:
    103: $chain->rewind();
    104: while ($chain->valid()) {
    105: $node = $chain->current();
    106: if (!$node->getRule() instanceof MultiConflictRule) {
    107: $otherWatch = $node->getOtherWatch($literal);
    108:
    109: if (!$node->getRule()->isDisabled() && !$decisions->satisfy($otherWatch)) {
    110: $ruleLiterals = $node->getRule()->getLiterals();
    111:
    112: $alternativeLiterals = array_filter($ruleLiterals, static function ($ruleLiteral) use ($literal, $otherWatch, $decisions): bool {
    113: return $literal !== $ruleLiteral &&
    114: $otherWatch !== $ruleLiteral &&
    115: !$decisions->conflict($ruleLiteral);
    116: });
    117:
    118: if (\count($alternativeLiterals) > 0) {
    119: reset($alternativeLiterals);
    120: $this->moveWatch($literal, current($alternativeLiterals), $node);
    121: continue;
    122: }
    123:
    124: if ($decisions->conflict($otherWatch)) {
    125: return $node->getRule();
    126: }
    127:
    128: $decisions->decide($otherWatch, $level, $node->getRule());
    129: }
    130: } else {
    131: foreach ($node->getRule()->getLiterals() as $otherLiteral) {
    132: if ($literal !== $otherLiteral && !$decisions->satisfy($otherLiteral)) {
    133: if ($decisions->conflict($otherLiteral)) {
    134: return $node->getRule();
    135: }
    136:
    137: $decisions->decide($otherLiteral, $level, $node->getRule());
    138: }
    139: }
    140: }
    141:
    142: $chain->next();
    143: }
    144:
    145: return null;
    146: }
    147:
    148: /**
    149: * Moves a rule node from one watch chain to another
    150: *
    151: * The rule node's watched literals are updated accordingly.
    152: *
    153: * @param int $fromLiteral A literal the node used to watch
    154: * @param int $toLiteral A literal the node should watch now
    155: * @param RuleWatchNode $node The rule node to be moved
    156: */
    157: protected function moveWatch(int $fromLiteral, int $toLiteral, RuleWatchNode $node): void
    158: {
    159: if (!isset($this->watchChains[$toLiteral])) {
    160: $this->watchChains[$toLiteral] = new RuleWatchChain;
    161: }
    162:
    163: $node->moveWatch($fromLiteral, $toLiteral);
    164: $this->watchChains[$fromLiteral]->remove();
    165: $this->watchChains[$toLiteral]->unshift($node);
    166: }
    167: }
