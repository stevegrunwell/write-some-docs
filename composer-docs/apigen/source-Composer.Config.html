    170: /**
    171: * Merges new config values with the existing ones (overriding)
    172: *
    173: * @param array{config?: array<string, mixed>, repositories?: array<mixed>} $config
    174: */
    175: public function merge(array $config, string $source = self::SOURCE_UNKNOWN): void
    176: {
    177: // override defaults with given config
    178: if (!empty($config['config']) && is_array($config['config'])) {
    179: foreach ($config['config'] as $key => $val) {
    180: if (in_array($key, ['bitbucket-oauth', 'github-oauth', 'gitlab-oauth', 'gitlab-token', 'http-basic', 'bearer'], true) && isset($this->config[$key])) {
    181: $this->config[$key] = array_merge($this->config[$key], $val);
    182: $this->setSourceOfConfigValue($val, $key, $source);
    183: } elseif (in_array($key, ['allow-plugins'], true) && isset($this->config[$key]) && is_array($this->config[$key]) && is_array($val)) {
    184: // merging $val first to get the local config on top of the global one, then appending the global config,
    185: // then merging local one again to make sure the values from local win over global ones for keys present in both
    186: $this->config[$key] = array_merge($val, $this->config[$key], $val);
    187: $this->setSourceOfConfigValue($val, $key, $source);
    188: } elseif (in_array($key, ['gitlab-domains', 'github-domains'], true) && isset($this->config[$key])) {
    189: $this->config[$key] = array_unique(array_merge($this->config[$key], $val));
    190: $this->setSourceOfConfigValue($val, $key, $source);
    191: } elseif ('preferred-install' === $key && isset($this->config[$key])) {
    192: if (is_array($val) || is_array($this->config[$key])) {
    193: if (is_string($val)) {
    194: $val = ['*' => $val];
    195: }
    196: if (is_string($this->config[$key])) {
    197: $this->config[$key] = ['*' => $this->config[$key]];
    198: $this->sourceOfConfigValue[$key . '*'] = $source;
    199: }
    200: $this->config[$key] = array_merge($this->config[$key], $val);
    201: $this->setSourceOfConfigValue($val, $key, $source);
    202: // the full match pattern needs to be last
    203: if (isset($this->config[$key]['*'])) {
    204: $wildcard = $this->config[$key]['*'];
    205: unset($this->config[$key]['*']);
    206: $this->config[$key]['*'] = $wildcard;
    207: }
    208: } else {
    209: $this->config[$key] = $val;
    210: $this->setSourceOfConfigValue($val, $key, $source);
    211: }
    212: } elseif ('audit' === $key) {
    213: $currentIgnores = $this->config['audit']['ignore'];
    214: $this->config[$key] = array_merge($this->config['audit'], $val);
    215: $this->setSourceOfConfigValue($val, $key, $source);
    216: $this->config['audit']['ignore'] = array_merge($currentIgnores, $val['ignore'] ?? []);
    217: } else {
    218: $this->config[$key] = $val;
    219: $this->setSourceOfConfigValue($val, $key, $source);
    220: }
    221: }
    222: }
    223:
    224: if (!empty($config['repositories']) && is_array($config['repositories'])) {
    225: $this->repositories = array_reverse($this->repositories, true);
    226: $newRepos = array_reverse($config['repositories'], true);
    227: foreach ($newRepos as $name => $repository) {
    228: // disable a repository by name
    229: if (false === $repository) {
    230: $this->disableRepoByName((string) $name);
    231: continue;
    232: }
    233:
    234: // disable a repository with an anonymous {"name": false} repo
    235: if (is_array($repository) && 1 === count($repository) && false === current($repository)) {
    236: $this->disableRepoByName((string) key($repository));
    237: continue;
    238: }
    239:
    240: // auto-deactivate the default packagist.org repo if it gets redefined
    241: if (isset($repository['type'], $repository['url']) && $repository['type'] === 'composer' && Preg::isMatch('{^https?://(?:[a-z0-9-.]+\.)?packagist.org(/|$)}', $repository['url'])) {
    242: $this->disableRepoByName('packagist.org');
    243: }
    244:
    245: // store repo
    246: if (is_int($name)) {
    247: $this->repositories[] = $repository;
    248: $this->setSourceOfConfigValue($repository, 'repositories.' . array_search($repository, $this->repositories, true), $source);
    249: } else {
    250: if ($name === 'packagist') { // BC support for default "packagist" named repo
    251: $this->repositories[$name . '.org'] = $repository;
    252: $this->setSourceOfConfigValue($repository, 'repositories.' . $name . '.org', $source);
    253: } else {
    254: $this->repositories[$name] = $repository;
    255: $this->setSourceOfConfigValue($repository, 'repositories.' . $name, $source);
    256: }
    257: }
    258: }
    259: $this->repositories = array_reverse($this->repositories, true);
    260: }

    271: /**
    272: * Returns a setting
    273: *
    274: * @param int $flags Options (see class constants)
    275: * @throws \RuntimeException
    276: *
    277: * @return mixed
    278: */
    279: public function get(string $key, int $flags = 0)
    280: {
    281: switch ($key) {
    282: // strings/paths with env var and {$refs} support
    283: case 'vendor-dir':
    284: case 'bin-dir':
    285: case 'process-timeout':
    286: case 'data-dir':
    287: case 'cache-dir':
    288: case 'cache-files-dir':
    289: case 'cache-repo-dir':
    290: case 'cache-vcs-dir':
    291: case 'cafile':
    292: case 'capath':
    293: // convert foo-bar to COMPOSER_FOO_BAR and check if it exists since it overrides the local config
    294: $env = 'COMPOSER_' . strtoupper(strtr($key, '-', '_'));
    295:
    296: $val = $this->getComposerEnv($env);
    297: if ($val !== false) {
    298: $this->setSourceOfConfigValue($val, $key, $env);
    299: }
    300:
    301: if ($key === 'process-timeout') {
    302: return max(0, false !== $val ? (int) $val : $this->config[$key]);
    303: }
    304:
    305: $val = rtrim((string) $this->process(false !== $val ? $val : $this->config[$key], $flags), '/\\');
    306: $val = Platform::expandPath($val);
    307:
    308: if (substr($key, -4) !== '-dir') {
    309: return $val;
    310: }
    311:
    312: return (($flags & self::RELATIVE_PATHS) === self::RELATIVE_PATHS) ? $val : $this->realpath($val);
    313:
    314: // booleans with env var support
    315: case 'cache-read-only':
    316: case 'htaccess-protect':
    317: // convert foo-bar to COMPOSER_FOO_BAR and check if it exists since it overrides the local config
    318: $env = 'COMPOSER_' . strtoupper(strtr($key, '-', '_'));
    319:
    320: $val = $this->getComposerEnv($env);
    321: if (false === $val) {
    322: $val = $this->config[$key];
    323: } else {
    324: $this->setSourceOfConfigValue($val, $key, $env);
    325: }
    326:
    327: return $val !== 'false' && (bool) $val;
    328:
    329: // booleans without env var support
    330: case 'disable-tls':
    331: case 'secure-http':
    332: case 'use-github-api':
    333: case 'lock':
    334: // special case for secure-http
    335: if ($key === 'secure-http' && $this->get('disable-tls') === true) {
    336: return false;
    337: }
    338:
    339: return $this->config[$key] !== 'false' && (bool) $this->config[$key];
    340:
    341: // ints without env var support
    342: case 'cache-ttl':
    343: return max(0, (int) $this->config[$key]);
    344:
    345: // numbers with kb/mb/gb support, without env var support
    346: case 'cache-files-maxsize':
    347: if (!Preg::isMatch('/^\s*([0-9.]+)\s*(?:([kmg])(?:i?b)?)?\s*$/i', (string) $this->config[$key], $matches)) {
    348: throw new \RuntimeException(
    349: "Could not parse the value of '$key': {$this->config[$key]}"
    350: );
    351: }
    352: $size = (float) $matches[1];
    353: if (isset($matches[2])) {
    354: switch (strtolower($matches[2])) {
    355: case 'g':
    356: $size *= 1024;
    357: // intentional fallthrough
    358: // no break
    359: case 'm':
    360: $size *= 1024;
    361: // intentional fallthrough
    362: // no break
    363: case 'k':
    364: $size *= 1024;
    365: break;
    366: }
    367: }
    368:
    369: return max(0, (int) $size);
    370:
    371: // special cases below
    372: case 'cache-files-ttl':
    373: if (isset($this->config[$key])) {
    374: return max(0, (int) $this->config[$key]);
    375: }
    376:
    377: return $this->get('cache-ttl');
    378:
    379: case 'home':
    380: return rtrim($this->process(Platform::expandPath($this->config[$key]), $flags), '/\\');
    381:
    382: case 'bin-compat':
    383: $value = $this->getComposerEnv('COMPOSER_BIN_COMPAT') ?: $this->config[$key];
    384:
    385: if (!in_array($value, ['auto', 'full', 'proxy', 'symlink'])) {
    386: throw new \RuntimeException(
    387: "Invalid value for 'bin-compat': {$value}. Expected auto, full or proxy"
    388: );
    389: }
    390:
    391: if ($value === 'symlink') {
    392: trigger_error('config.bin-compat "symlink" is deprecated since Composer 2.2, use auto, full (for Windows compatibility) or proxy instead.', E_USER_DEPRECATED);
    393: }
    394:
    395: return $value;
    396:
    397: case 'discard-changes':
    398: $env = $this->getComposerEnv('COMPOSER_DISCARD_CHANGES');
    399: if ($env !== false) {
    400: if (!in_array($env, ['stash', 'true', 'false', '1', '0'], true)) {
    401: throw new \RuntimeException(
    402: "Invalid value for COMPOSER_DISCARD_CHANGES: {$env}. Expected 1, 0, true, false or stash"
    403: );
    404: }
    405: if ('stash' === $env) {
    406: return 'stash';
    407: }
    408:
    409: // convert string value to bool
    410: return $env !== 'false' && (bool) $env;
    411: }
    412:
    413: if (!in_array($this->config[$key], [true, false, 'stash'], true)) {
    414: throw new \RuntimeException(
    415: "Invalid value for 'discard-changes': {$this->config[$key]}. Expected true, false or stash"
    416: );
    417: }
    418:
    419: return $this->config[$key];
    420:
    421: case 'github-protocols':
    422: $protos = $this->config['github-protocols'];
    423: if ($this->config['secure-http'] && false !== ($index = array_search('git', $protos))) {
    424: unset($protos[$index]);
    425: }
    426: if (reset($protos) === 'http') {
    427: throw new \RuntimeException('The http protocol for github is not available anymore, update your config\'s github-protocols to use "https", "git" or "ssh"');
    428: }
    429:
    430: return $protos;
    431:
    432: case 'autoloader-suffix':
    433: if ($this->config[$key] === '') { // we need to guarantee null or non-empty-string
    434: return null;
    435: }
    436:
    437: return $this->process($this->config[$key], $flags);
    438:
    439: default:
    440: if (!isset($this->config[$key])) {
    441: return null;
    442: }
    443:
    444: return $this->process($this->config[$key], $flags);
    445: }
    446: }

    563: /**
    564: * Validates that the passed URL is allowed to be used by current config, or throws an exception.
    565: *
    566: * @param IOInterface $io
    567: * @param mixed[] $repoOptions
    568: */
    569: public function prohibitUrlByConfig(string $url, ?IOInterface $io = null, array $repoOptions = []): void
    570: {
    571: // Return right away if the URL is malformed or custom (see issue #5173)
    572: if (false === filter_var($url, FILTER_VALIDATE_URL)) {
    573: return;
    574: }
    575:
    576: // Extract scheme and throw exception on known insecure protocols
    577: $scheme = parse_url($url, PHP_URL_SCHEME);
    578: $hostname = parse_url($url, PHP_URL_HOST);
    579: if (in_array($scheme, ['http', 'git', 'ftp', 'svn'])) {
    580: if ($this->get('secure-http')) {
    581: if ($scheme === 'svn') {
    582: if (in_array($hostname, $this->get('secure-svn-domains'), true)) {
    583: return;
    584: }
    585:
    586: throw new TransportException("Your configuration does not allow connections to $url. See https://getcomposer.org/doc/06-config.md#secure-svn-domains for details.");
    587: }
    588:
    589: throw new TransportException("Your configuration does not allow connections to $url. See https://getcomposer.org/doc/06-config.md#secure-http for details.");
    590: }
