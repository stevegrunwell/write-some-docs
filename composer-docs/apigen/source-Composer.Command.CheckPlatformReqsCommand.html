    33: new InputOption('lock', null, InputOption::VALUE_NONE, 'Checks requirements only from the lock file, not from installed packages.'),

    49: {
    50: $composer = $this->requireComposer();
    51:
    52: $requires = [];
    53: $removePackages = [];

    55: $this->getIO()->writeError('<info>Checking '.($input->getOption('no-dev') ? 'non-dev ' : '').'platform requirements using the lock file</info>');
    56: $installedRepo = $composer->getLocker()->getLockedRepository(!$input->getOption('no-dev'));
    57: } else {
    58: $installedRepo = $composer->getRepositoryManager()->getLocalRepository();
    59: // fallback to lockfile if installed repo is empty
    60: if (!$installedRepo->getPackages()) {
    61: $this->getIO()->writeError('<warning>No vendor dir present, checking '.($input->getOption('no-dev') ? 'non-dev ' : '').'platform requirements from the lock file</warning>');
    62: $installedRepo = $composer->getLocker()->getLockedRepository(!$input->getOption('no-dev'));
    63: } else {
    64: if ($input->getOption('no-dev')) {
    65: $removePackages = $installedRepo->getDevPackageNames();
    66: }
    67:
    68: $this->getIO()->writeError('<info>Checking '.($input->getOption('no-dev') ? 'non-dev ' : '').'platform requirements for packages in the vendor dir</info>');
    69: }
    70: }
    71: if (!$input->getOption('no-dev')) {
    72: foreach ($composer->getPackage()->getDevRequires() as $require => $link) {
    73: $requires[$require] = [$link];
    74: }
    75: }
    76:
    77: $installedRepo = new InstalledRepository([$installedRepo, new RootPackageRepository(clone $composer->getPackage())]);
    78: foreach ($installedRepo->getPackages() as $package) {
    79: if (in_array($package->getName(), $removePackages, true)) {
    80: continue;
    81: }

    83: $requires[$require][] = $link;
    84: }
    85: }

    91: $results = [];

    95: * @var Link[] $links
    96: */

    98: if (PlatformRepository::isPlatformPackage($require)) {

    104: if ($candidate->getName() === $require) {
    105: $candidateConstraint = new Constraint('=', $candidate->getVersion());

    107: } else {
    108: foreach (array_merge($candidate->getProvides(), $candidate->getReplaces()) as $link) {

    112: }
    113: }
    114: }
    115:
    116: // safety check for phpstan, but it should not be possible to get a candidate out of findPackagesWithReplacersAndProviders without a constraint matching $require
    117: if (!$candidateConstraint) {
    118: continue;
    119: }
    120:

    122: if (!$link->getConstraint()->matches($candidateConstraint)) {

    125: $candidateConstraint->getPrettyString(),
    126: $link,
    127: '<error>failed</error>',
    128: $candidate->getName() === $require ? '' : '<comment>provided by '.$candidate->getPrettyName().'</comment>',
    129: ];
    130:
    131: // skip to next candidate
    132: continue 2;
    133: }
    134: }
    135:
    136: $results[] = [
    137: $candidate->getName() === $require ? $candidate->getPrettyName() : $require,
    138: $candidateConstraint->getPrettyString(),
    139: null,
    140: '<info>success</info>',
    141: $candidate->getName() === $require ? '' : '<comment>provided by '.$candidate->getPrettyName().'</comment>',
    142: ];
    143:
    144: // candidate matched, skip to next requirement
    145: continue 2;
    146: }
    147:
    148: // show the first error from every failed candidate
    149: $results = array_merge($results, $reqResults);
    150: $exitCode = max($exitCode, 1);
    151:
    152: continue;
    153: }
    154:
    155: $results[] = [
    156: $require,
    157: 'n/a',
    158: $links[0],
    159: '<error>missing</error>',
    160: '',
    161: ];
    162:
    163: $exitCode = max($exitCode, 2);
    164: }
    165: }
    166:

    176: {
    177: $rows = [];

    179: /**

    184: if ('json' === $format) {

    186: "name" => $platformPackage,
    187: "version" => $version,

    190: 'source' => $link->getSource(),
    191: 'type' => $link->getDescription(),

    195: "provider" => $provider === '' ? null : strip_tags($provider),

    203: rtrim($status.' '.$provider),

    205: }
    206: }

    208: if ('json' === $format) {
