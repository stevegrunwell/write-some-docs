    200: * @param string[]|string $messages

    202: private function doOverwrite($messages, bool $newline, ?int $size, bool $stderr, int $verbosity): void
    203: {
    204: // messages can be an array, let's convert it to string anyway
    205: $messages = implode($newline ? "\n" : '', (array) $messages);
    206:
    207: // since overwrite is supposed to overwrite last message...
    208: if (!isset($size)) {
    209: // removing possible formatting of lastMessage with strip_tags
    210: $size = strlen(strip_tags($stderr ? $this->lastMessageErr : $this->lastMessage));
    211: }
    212: // ...let's fill its length with backspaces
    213: $this->doWrite(str_repeat("\x08", $size), false, $stderr, $verbosity);
    214:
    215: // write the new message
    216: $this->doWrite($messages, false, $stderr, $verbosity);
    217:
    218: // In cmd.exe on Win8.1 (possibly 10?), the line can not be cleared, so we need to
    219: // track the length of previous output and fill it with spaces to make sure the line is cleared.
    220: // See https://github.com/composer/composer/pull/5836 for more details
    221: $fill = $size - strlen(strip_tags($messages));
    222: if ($fill > 0) {
    223: // whitespace whatever has left
    224: $this->doWrite(str_repeat(' ', $fill), false, $stderr, $verbosity);
    225: // move the cursor back
    226: $this->doWrite(str_repeat("\x08", $fill), false, $stderr, $verbosity);
    227: }
    228:
    229: if ($newline) {
    230: $this->doWrite('', true, $stderr, $verbosity);
    231: }
    232:
    233: if ($stderr) {
    234: $this->lastMessageErr = $messages;
    235: } else {
    236: $this->lastMessage = $messages;
    237: }
